4   import java.util.Scanner;
5   
6   
7   public class Pelikierros {
8       private Scanner lukija;
9       private Pelaaja pelaaja;
10      private TehtavanArpoja arpoja;
11      private int pisteet;
12      
13      
14      public Pelikierros(Pelaaja pelaaja) {
15          this.lukija = new Scanner(System.in);
16          this.pelaaja = pelaaja;
17          this.arpoja = new TehtavanArpoja(pelaaja.getTaso());
18          this.pisteet = 0;
19      }
20      
21      public void pelaa() {
22          for (int i = 0; i < 10; i++) {
23              int luku1 = arpoja.haeLuku();
24              int luku2 = arpoja.haeLuku();
25              System.out.println("Paljonko on " + arpoja.tehtava(luku1, luku2) + "?");
26              int vastaus = Integer.parseInt(lukija.nextLine());
27              Tarkastaja tarkastaja = new Tarkastaja(vastaus, arpoja.lasku(luku1, luku2));
28              if (tarkastaja.onkoOikein()) {
29                  pisteet++;
30              } else {
31                  System.out.println("Väärin, oikea vastaus on " + tarkastaja.oikeaVastaus());
32              }
33          }
34          System.out.println("Sait " + pisteet + "/10 pistettä");
35          lopetaKierros();
36      }
37      
38      public void lopetaKierros() {
39          int tahdet = 0;
40          if (pisteet < 5) {
41              System.out.println("Et läpäissyt tasoa. Yritä uudelleen (k/e)?");
42          } else {
43              tahdet = (pisteet - 5) / 2 + 1;
44              System.out.println("Taso suoritettu" + tahdet + " tähdellä.\nJatketaanko seuraavalle tasolle (k/e)?");
45              pelaaja.setTaso(pelaaja.getTaso() + 1);
46          }
47          String syote = lukija.nextLine();
48          pisteet = 0;
49          if (syote.equals("k")) {
50              this.arpoja = new TehtavanArpoja(pelaaja.getTaso());
51              pelaa();
52          }
53      }
54      
55      public int haePisteet() {
56          return this.pisteet;
57      }
58  }
